    cdf_aux_mod.f90
    cdf_beta_mod.f90

    biomath_mathlib_mod.f90

    zero_finder.f90

    biomath_mathlib_mod.f90

    cdf_f_mod.f90
    biomath_constants_mod.f90
    cdf_aux_mod.f90
    cdf_beta_mod.f90
    zero_finder.f90
    biomath_mathlib_mod.f90

    biomath_constants_mod.f90
    biomath_mathlib_mod.f90

    cdf_poisson_mod.f90
    biomath_constants_mod.f90
    cdf_aux_mod.f90
    cdf_gamma_mod.f90
    biomath_mathlib_mod.f90

  TARGETS
    biomath_constants_mod
    biomath_sort_mod
    biomath_strings_mod
    biomath_interface_mod
    biomath_mathlib_mod

    cdf_aux_mod
    cdf_beta_mod
